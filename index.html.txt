        let tg = window.Telegram.WebApp;

        function sendMoney(type) {
            let val = document.getElementById('money_val').value;
            if(!val) return;
            tg.sendData(JSON.stringify({action: type, amount: parseFloat(val)}));
            tg.close(); // Закрываем после отправки
        }

        function sendTask(type, name) {
            tg.sendData(JSON.stringify({action: 'task', task_type: type, task_name: name}));
            tg.close();
        }
