            <button class="btn-income" onclick="updateMoney('income')">Доход +</button>

    <script>
        // Инициализация Telegram WebApp
        let tg = window.Telegram.WebApp;
        tg.expand(); // Расширяем на весь экран

        // Игровые переменные (пока храним локально)
        let currentBalance = 0;
        let nextGoal = 10000; // Цель, при которой колба будет полной на 100%

        // DOM элементы
        const liquid = document.getElementById('liquid');
        const balanceDisplay = document.getElementById('balanceDisplay');
        const amountInput = document.getElementById('amountInput');

        // Функция обновления визуализации
        function updateFlaskView() {
            // Рассчитываем процент заполнения (не больше 100)
            let percentage = Math.min((currentBalance / nextGoal) * 100, 100);
            
            // Если баланс отрицательный, жидкость пропадает (но лучше не допускать)
            if (currentBalance < 0) percentage = 0;

            // Устанавливаем высоту жидкости
            liquid.style.height = percentage + '%';

            // Обновляем текст баланса
            balanceDisplay.innerText = currentBalance.toFixed(2) + ' ₽';

            // Логика цвета: если баланс падает, можно менять цвет на оранжевый (по желанию)
            if (currentBalance < 1000) {
                liquid.style.backgroundColor = '#f0883e'; // Предупреждение
            } else {
                liquid.style.backgroundColor = '#238636'; // Норма
            }
        }

        // Функция для кнопок (Тест)
        function updateMoney(type) {
            let amount = parseFloat(amountInput.value);
            if (isNaN(amount) || amount <= 0) return;

            if (type === 'income') {
                currentBalance += amount;
            } else if (type === 'expense') {
                currentBalance -= amount;
            }

            amountInput.value = ''; // Очистить поле
            updateFlaskView();
        }

        // Генерируем пузырьки для красоты
        function createBubbles() {
            for (let i = 0; i < 10; i++) {
                let bubble = document.createElement('div');
                bubble.className = 'bubble';
                // Случайный размер, задержка и позиция
                let size = Math.random() * 8 + 4 + 'px';
                bubble.style.width = size;
                bubble.style.height = size;
                bubble.style.left = Math.random() * 100 + '%';
                bubble.style.animationDelay = Math.random() * 4 + 's';
                bubble.style.animationDuration = Math.random() * 3 + 2 + 's';
                liquid.appendChild(bubble);
            }
        }

        // Первичный запуск
        updateFlaskView();
        createBubbles();

    </script>
